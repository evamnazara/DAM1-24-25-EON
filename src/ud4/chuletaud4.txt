    public static void desordenar(int t[]) {
            Random rnd = new Random();

            for (int i = 0; i < t.length; i++) {
                int nuevaPosicion = rnd.nextInt(t.length);
                
                int auxiliar = t[i];

                t[i] = t[nuevaPosicion];
                t[nuevaPosicion] = auxiliar;

            }
        }
